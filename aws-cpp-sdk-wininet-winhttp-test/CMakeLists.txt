cmake_minimum_required(VERSION 2.6)
project(aws-cpp-sdk-dynamodb-wininet-winhttp-test)

# Headers are included in the source so that they show up in Visual Studio.
# They are included elsewhere for consistency.

file(GLOB AWS_DYNAMO_SRC "${CMAKE_CURRENT_SOURCE_DIR}/*.cpp")
file(GLOB AWS_DYNAMO_INTEGRATION_TESTS_SRC ${AWS_DYNAMO_SRC})

if(NOT MSVC)
    # With gcc 4.9 onwards this should be -std=c++11 or -std=c++14
    add_definitions(-std=c++0x)
endif()


set(AWS_DYNAMO_INTEGRATION_TEST_APPLICATION_INCLUDES
  "${AWS_NATIVE_SDK_ROOT}/aws-cpp-sdk-core/include/"
  "${AWS_NATIVE_SDK_ROOT}/aws-cpp-sdk-dynamodb/include/"
  "${AWS_NATIVE_SDK_ROOT}/testing-resources/include/"
)

include_directories(${AWS_DYNAMO_INTEGRATION_TEST_APPLICATION_INCLUDES})

if(MSVC AND BUILD_SHARED_LIBS)
    add_definitions(-DGTEST_LINKED_AS_SHARED_LIBRARY=1)
endif()

enable_testing()
add_executable(runInetHttpTest ${AWS_DYNAMO_INTEGRATION_TESTS_SRC})
if(NOT MSVC)
    target_link_libraries(runInetHttpTest aws-cpp-sdk-dynamodb testing-resources psapi)
else()
    target_link_libraries(runInetHttpTest aws-cpp-sdk-dynamodb testing-resources)
endif()
copyDlls(runInetHttpTest aws-cpp-sdk-core aws-cpp-sdk-dynamodb testing-resources)
